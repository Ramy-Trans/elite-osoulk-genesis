  try {
    const results = await Promise.allSettled([
      db.getUsers?.(),
      db.getListings?.(),
      db.getLeads?.(),
      db.getViews?.()
    ]);

    const safe = (r, fallback = []) =>
      r.status === "fulfilled" ? r.value : fallback;

    const users = safe(results[0], []);
    const listings = safe(results[1], []);
    const leads = safe(results[2], []);
    const views = safe(results[3], {});

    const dbHealthy = results.every(r => r.status === "fulfilled");

    return res.status(dbHealthy ? 200 : 206).json({
      ok: dbHealthy,
      warning: !dbHealthy ? "Partial DB failure" : null,
      data: {
        users,
        listings,
        leads,
        views
      }
    });

  } catch (err) {
    console.error("[stats fatal]", err);
    return res.status(500).json({
      ok: false,
      error: "stats failed completely"
    });
  }
});
